    function createCodeBlock(codeText, btnLabel = "📋 نسخ الأمر") {
        const container = document.createElement("div");
        container.style.marginBottom = "0.8rem";
        const header = document.createElement("div");
        header.className = "code-header";
        const copyBtn = document.createElement("button");
        copyBtn.className = "copy-btn";
        copyBtn.innerHTML = btnLabel;
        copyBtn.onclick = () => copyToClipboard(codeText, copyBtn);
        header.appendChild(copyBtn);
        const pre = document.createElement("pre");
        const code = document.createElement("code");
        code.innerText = codeText;
        pre.appendChild(code);
        container.appendChild(header);
        container.appendChild(pre);
        return container;
    }

    function copyToClipboard(text, btnElement) {
        navigator.clipboard.writeText(text).then(() => {
            const original = btnElement.innerHTML;
            btnElement.innerHTML = "✅ تم النسخ!";
            setTimeout(() => { btnElement.innerHTML = original; }, 1800);
        }).catch(() => alert("فشل النسخ، حاول يدوياً."));
    }

    function renderSteps() {
        const container = document.getElementById("stepsContainer");
        container.innerHTML = "";
        steps.forEach((step, idx) => {
            const card = document.createElement("div");
            card.className = "step-card";

            const header = document.createElement("div");
            header.className = "step-header";

            const left = document.createElement("div");
            left.style.display = "flex";
            left.style.alignItems = "center";
            const numSpan = document.createElement("span");
            numSpan.className = "step-num";
            numSpan.innerText = idx + 1;
            const titleSpan = document.createElement("span");
            titleSpan.className = "step-title";
            titleSpan.innerText = step.title;
            left.appendChild(numSpan);
            left.appendChild(titleSpan);

            const copyHeaderBtn = document.createElement("button");
            copyHeaderBtn.className = "copy-btn";
            copyHeaderBtn.innerHTML = "📋 نسخ هذه الخطوة";
            copyHeaderBtn.onclick = () => {
                copyToClipboard(`## ${step.title}\n\n${step.content}`, copyHeaderBtn);
            };
            header.appendChild(left);
            header.appendChild(copyHeaderBtn);

            const contentDiv = document.createElement("div");
            contentDiv.className = "step-content";

            // معالجة المحتوى: استخراج الأوامر المحاطة بـ ``` أو التي تبدأ بـ $
            let raw = step.content;
            let lines = raw.split('\n');
            let inCode = false;
            let codeBuffer = "";
            let resultHtml = "";

            for (let line of lines) {
                if (line.trim().startsWith("```")) {
                    if (inCode) {
                        // نهاية كتلة كود
                        if (codeBuffer.trim()) {
                            const codeBlockDiv = createCodeBlock(codeBuffer.trim(), "📋 نسخ الكتلة");
                            resultHtml += codeBlockDiv.outerHTML;
                            codeBuffer = "";
                        }
                        inCode = false;
                    } else {
                        inCode = true;
                    }
                    continue;
                }
                if (inCode) {
                    codeBuffer += line + "\n";
                    continue;
                }
                // الأوامر التي تبدأ بـ $ أو علامة backtick
                let trimmed = line.trim();
                if (trimmed.startsWith("$ ")) {
                    let cmd = trimmed.substring(2);
                    const tempDiv = document.createElement("div");
                    tempDiv.style.margin = "0.5rem 0";
                    const btn = document.createElement("button");
                    btn.className = "copy-btn";
                    btn.style.marginRight = "0.5rem";
                    btn.innerHTML = "📋 نسخ الأمر";
                    btn.onclick = () => copyToClipboard(cmd, btn);
                    const codeSpan = document.createElement("code");
                    codeSpan.className = "inline-code";
                    codeSpan.innerText = cmd;
                    tempDiv.appendChild(btn);
                    tempDiv.appendChild(codeSpan);
                    resultHtml += tempDiv.outerHTML;
                } else {
                    resultHtml += line + "<br>";
                }
            }
            if (codeBuffer.trim()) {
                const codeBlockDiv = createCodeBlock(codeBuffer.trim(), "📋 نسخ الكتلة");
                resultHtml += codeBlockDiv.outerHTML;
            }
            // إضافة زر نسخ النص الكامل للخطوة
            const fullCopyBtn = document.createElement("button");
            fullCopyBtn.className = "copy-btn";
            fullCopyBtn.style.marginTop = "0.8rem";
            fullCopyBtn.innerHTML = "📋 نسخ كل محتوى الخطوة (نصي)";
            fullCopyBtn.onclick = () => copyToClipboard(step.content.replace(/`/g, '').replace(/\$ /g, ''), fullCopyBtn);
            resultHtml += "<hr>";
            resultHtml += fullCopyBtn.outerHTML;

            contentDiv.innerHTML = resultHtml;
            card.appendChild(header);
            card.appendChild(contentDiv);
            container.appendChild(card);
        });
    }

    function getFullPrompt() {
        let full = "# برومبت هندسي متكامل – خطة اختراق الاشتراك المدفوع وسحب البيانات وإرسالها إلى تلغرام\n\n";
        full += "**الهدف الأكاديمي:** تحويل الخطة المدفوعة (Pro) إلى مجانية، وسحب قاعدة البيانات السحابية، ثم إرسال البيانات إلى بوت تلغرام، باستخدام مفاتيح وعناوين مستخرجة من APK.\n\n";
        full += "## التسلسل العملي الكامل:\n\n";
        steps.forEach((s, i) => {
            full += `### ${i+1}. ${s.title}\n${s.content.replace(/`/g, '`').replace(/\*/g, '•')}\n\n`;
        });
        full += "\n**ملاحظة:** هذا البرومبت مخصص للتدريب العملي في مجال الهندسة العكسية والاختراق الأخلاقي ضمن بيئة مرخصة واختبارية.\n";
        return full;
    }

    renderSteps();

        copyToClipboard(getFullPrompt(), this);
