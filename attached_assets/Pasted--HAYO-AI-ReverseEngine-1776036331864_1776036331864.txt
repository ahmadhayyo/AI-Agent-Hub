**4. Vulnerability Scoring System الحقيقي (الأولوية: عالية)**
```typescript
// الحالي: ThreatGauge يستخدم نقاط عشوائية
// المطلوب: نقاط بناءً على نتائج حقيقية

const vulnScoringMatrix = {
  'Hardcoded API Key': { base: 25, category: 'critical' },
  'Unencrypted Communications': { base: 20, category: 'critical' },
  'Weak Cryptography': { base: 15, category: 'high' },
  'Excessive Permissions': { base: 10, category: 'high' },
  'Debuggable APK': { base: 12, category: 'high' },
  'Outdated Libraries': { base: 8, category: 'medium' },
  'Log Statements with Sensitive Data': { base: 5, category: 'medium' },
};

function calculateRealVulnScore(findings: VulnerabilityFinding[]): number {
  const score = findings.reduce((sum, f) => {
    const rule = Object.entries(vulnScoringMatrix).find(
      ([rule]) => f.title.includes(rule)
    );
    return sum + (rule ? rule[1].base : 2);
  }, 0);
  
  // عامل معدِّل بناءً على التكرار والخطورة
  const criticalCount = findings.filter(f => f.severity === 'critical').length;
  const multiplier = Math.min(1.5, 1 + (criticalCount * 0.1));
  
  return Math.min(100, Math.round(score * multiplier));
}
```

```typescript
  if (!smartInst.trim()) return;

  //   scope: 'file' | 'method' | 'class' | 'package'
```

**2. نظام Undo/Redo الحقيقي**

```typescript
interface EditHistory {
  stateId: string;
  timestamp: number;
  changes: Map<string, string>; // filepath -> content
  description: string;
}

const [editHistory, setEditHistory] = useState<EditHistory[]>([]);
const [currentStateIndex, setCurrentStateIndex] = useState(0);

function saveEditState(description: string) {
  const newState: EditHistory = {
    stateId: `state_${Date.now()}`,
    timestamp: Date.now(),
    changes: new Map(eCache),
    description,
  };
  
  // حذف أي حالات "redo" إذا عدّلنا بعد Undo
  const newHistory = editHistory.slice(0, currentStateIndex + 1);
  newHistory.push(newState);
  
  setEditHistory(newHistory);
  setCurrentStateIndex(newHistory.length - 1);
}

function undo() {
  if (currentStateIndex > 0) {
    const previousState = editHistory[currentStateIndex - 1];
    eCache.clear();
    previousState.changes.forEach((content, path) => {
      eCache.set(path, content);
    });
    setCurrentStateIndex(currentStateIndex - 1);
    setEContent(eCache.get(eNode?.path || '') || '');
    toast.info(`↶ تم العودة إلى: ${previousState.description}`);
  }
}

function redo() {
  if (currentStateIndex < editHistory.length - 1) {
    const nextState = editHistory[currentStateIndex + 1];
    eCache.clear();
    nextState.changes.forEach((content, path) => {
      eCache.set(path, content);
    });
    setCurrentStateIndex(currentStateIndex + 1);
    setEContent(eCache.get(eNode?.path || '') || '');
    toast.info(`↷ تم إعادة: ${nextState.description}`);
  }
}
```

```typescript
// دالة مساعدة لاستخراج مفاتيح AWS
function extractAwsCredentialsFromAPK(apkFiles: DecompiledFile[]) {
  const results = [];
  
  for (const file of apkFiles) {
    if (!file.content) continue;
    
    // AWS Access Key pattern
    const accessKeys = file.content.match(/AKIA[0-9A-Z]{16}/g);
    if (accessKeys) {
      results.push({
        type: 'AWS Access Key',
        value: accessKeys[0],
        file: file.path,
        severity: 'critical',
      });
    }
    
    // AWS Secret Key pattern (أطول ويبدأ بـ wJ)
    const secretKeys = file.content.match(/wJ[a-zA-Z0-9\/+=]{40}/g);
    if (secretKeys) {
      results.push({
        type: 'AWS Secret Key',
        value: secretKeys[0],
        file: file.path,
        severity: 'critical',
      });
    }
  }
  
  return results;
}
```

```typescript
    .filter((s: any) => s.string.length > 3) // تجاهل النصوص القصيرة
    .map((s: any) => {

      return {
        original: s.string,

        type: encodingType,
        location: {

          lineNumber: s.address,

    });
```

```typescript
    setFXrefLoading(false);
  }
}

// Backend:
async function findXrefs(sessionId: string, query: string) {
  const tmpDir = sessionCache[sessionId];
  
  // استخدام JADX API أو Radare2
  const xrefs = [];
  
  // البحث في جميع الملفات Smali
  const smaliFiles = findFilesRecursive(tmpDir, ['.smali']);
  
  for (const file of smaliFiles) {
    const content = fs.readFileSync(file, 'utf-8');
    const lines = content.split('\n');
    
    lines.forEach((line, idx) => {
      if (line.includes(query)) {
        xrefs.push({
          file: file.replace(tmpDir, ''),
          lineNumber: idx + 1,
          content: line.trim(),
          type: line.includes('invoke') ? 'called_by' : 'reference',
        });
      }
    });
  }
  
  return { xrefs };
}
```

```typescript
// 4. استكشاف IDOR (Insecure Direct Object References)
async function phase4IDORExploit(apiEndpoints: string[], apiKeys: string[]) {
  const idorVulnerabilities = [];
  
  for (const endpoint of apiEndpoints) {
    for (const key of apiKeys) {
      try {
        // اختبر endpoints بنمط IDOR
        const variations = [
          endpoint.replace(/\d+/, '1'),
          endpoint.replace(/\{id\}/, '1'),
          endpoint.replace(/user_id=\d+/, 'user_id=1'),
        ];
        
        for (const variant of variations) {
          const response = await fetch(variant, {
            headers: { 'X-API-Key': key },
            timeout: 5000,
          });
          
          if (response.status === 200) {
            idorVulnerabilities.push({
              endpoint: variant,
              method: 'GET',
              accessLevel: 'Unauthorized',
              dataExposed: await response.json(),
            });
          }
        }
      } catch (e) {
        // timeout أو connection refused — تحقق من الخادم
      }
    }
  }
  
  return {
    phase: 4,
    idorVulnerabilities,
    criticalFindingsCount: idorVulnerabilities.length,
  };
}

  const steps = [];
  
  // Phase 1
  const phase1 = await phase1Decompile(apkBuffer);
  steps.push({ id: 1, title: 'Decompilation', ...phase1, status: 'success' });
  
  // Phase 2
  const phase2 = await phase2AuthAnalysis(phase1);

  const phase3 = await phase3KeyExtraction(phase1);
  steps.push({ id: 3, title: 'Key Extraction', ...phase3, status: phase3.foundKeys.apiKeys.length > 0 ? 'critical' : 'info' });
  
  // Phase 4-7 (اختياري إذا كانت هناك endpoints حقيقية)
  const endpoints = extractApiEndpoints(phase1.files);
  if (endpoints.length > 0 && phase3.foundKeys.apiKeys.length > 0) {
    // Phase 4
    const phase4 = await phase4IDORExploit(endpoints, phase3.foundKeys.apiKeys);
    steps.push({ id: 4, title: 'IDOR Testing', ...phase4, status: phase4.idorVulnerabilities.length > 0 ? 'critical' : 'success' });
  }
  
  // Phase 8
  const phase8 = await phase8ReportGeneration(steps);
  steps.push({ id: 8, title: 'Report Generation', ...phase8, status: 'success' });
  
  return {
    steps,
    summary: {
      extractedEndpoints: endpoints,
      criticalFindings: steps.filter(s => s.status === 'critical').length,
      totalVulnerabilities: steps.reduce((sum, s) => sum + (s.vulnerabilities?.length || 0), 0),
    },
    report: phase8.report,
  };
}
```
